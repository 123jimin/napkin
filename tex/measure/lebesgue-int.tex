\begin{definition}
	If $A$ is a measurable set of $\Omega$,
	then the \vocab{indicator function}
	$\mathbf{1}_A \colon \Omega \to \RR$ is defined by
	\[ \mathbf{1}_A(\omega) = \begin{cases}
			1 & \omega \in A \\
			0 & \omega \notin A.
		\end{cases} \]
\end{definition}
